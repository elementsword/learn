单例模式（Singleton Pattern）是设计模式里最常见的一种，核心思想是：
 **一个类在整个程序运行期间只能有一个实例，并且提供一个全局访问点。**

懒汉式

```cpp
#include <iostream>
#include <mutex>
using namespace std;


class Singleton{
        private:
                static Singleton * instance;
                static mutex mtx;
                Singleton(const Singleton&) =delete;
                Singleton & operator=(const Singleton&) =delete;
                Singleton(){
                        cout <<"Singleton create"<< endl;
                }

        public:
    //双重判断 
                static Singleton* getInstance(){
                        if(instance==nullptr){
                                lock_guard<mutex> lock(mtx);
                                if(instance==nullptr){
                                        instance = new Singleton();
                                }
                        }
                        return instance;
                }

                void show(){
                        cout<<"show " <<endl;
                }
};

Singleton * Singleton::instance =nullptr;
mutex Singleton::mtx;
int main(){
        Singleton::getInstance()->show();
}
```

饿汉式

```cpp
#include <iostream>


class Singleton{
        private:
                static Singleton *instance;
                Singleton(const Singleton &)=delete;
                Singleton &operator=(const Singleton &)=delete;
                Singleton(){
                        std::cout<<"create Singleton"<<std::endl;
                }

        public:
                static Singleton* getInstance(){
                        return instance;
                }
                void show(){
                        std::cout<<"show Singleton"<<std::endl;
                }



};

Singleton * Singleton::instance =new Singleton();

int main(){
        Singleton::getInstance()->show();
}
```
